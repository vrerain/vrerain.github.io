int BFSTraverse(GraphAdjList &g, int start)
{
    queue<int> q;
    int cnt = 0;
    int level = 0;
    int last = start;
    int tail = 0;
    int visited[g.numVertexs];
    for (int i = 0; i < g.numVertexs; ++i)
    {
        visited[i] = 0;
    }
    visited[start] = 1;
    q.push(start);
    cnt++;
    EdgeNode *te;
    while (!q.empty())
    {
        int temp = q.front();
        q.pop();
        te = g.adjList[temp].firstEdge;
        while(te != NULL)
        {
            if (!visited[te->adjvex-1])
            {
                q.push(te->adjvex - 1);
                visited[te->adjvex - 1] = 1;
                cnt++;
                tail = te->adjvex - 1;

            te = te->next;
        }
        if (temp == last)
        {
            last = tail;

            break;
        }
    }
    return cnt;
}
